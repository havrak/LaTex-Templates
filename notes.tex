%%% Tento soubor obsahuje základní konfiguraci a makra pro poznámky

%%% style pro header a footer na stránce s obsahem
\fancypagestyle{toc}{
  \fancyhf{}
  \renewcommand{\headrulewidth}{0.4pt}
  \renewcommand{\footrulewidth}{0.4pt}
  \fancyhead[C]{}
  \fancyhead[L]{\textbf{\title}}
  \fancyfoot[L]{\author}
  \fancyfoot[C]{}
  \fancyfoot[R]{\thepage}
}

%%% style pro header a footer pro běžné stránky
\fancypagestyle{plain}{
  \fancyhf{}
  \renewcommand{\headrulewidth}{0.4pt}
  \renewcommand{\footrulewidth}{0.4pt}
  \fancyhead[C]{}
  \fancyhead[L]{\textbf{\title \hspace{1mm}-- \thechapter. \leftmark}}
  \fancyfoot[L]{\author}
  \fancyfoot[C]{}
  \fancyfoot[R]{\thepage}
}

%%% Jednoduchá titulní stránka
\newcommand\titlep{
\pagestyle{empty}
\begin{center}
  \vspace{30mm}
  {\huge\title}
  \vspace{8mm}

  {\bf\Large\subtitle}
  \vfill
  Vypracoval: \hfill \author

  Škola: \hfill \institution
  \vspace{10mm}
\end{center}
\newpage

}

%%% Stránka s obsahem
\newcommand\tocp{\pagenumbering{roman}\addtocontents{toc}{\protect\thispagestyle{toc}}\pagestyle{toc}\tableofcontents\cleardoublepage}
