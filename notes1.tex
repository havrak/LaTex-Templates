%%% style pro header a footer pro běžné stránky
\fancypagestyle{plain}{
	\fancyhf{}
	\renewcommand{\headrulewidth}{0.4pt}
	\renewcommand{\footrulewidth}{0.4pt}
	\fancyhead[L]{\textbf{\title}}

	\fancyfoot[L]{\author}
	\fancyfoot[C]{}
	\fancyfoot[R]{\thepage}
}

  {\huge\title}
  \vspace{8mm}

  {\bf\Large\subtitle}
  \vfill
  Vypracoval: \hfill \author

  Instituce: \hfill \institution

	Vedl: \hfill \supervisor
  \vspace{10mm}
\end{center}
\newpage

}
