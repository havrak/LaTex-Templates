%%% Declarations for doc type notes

%%% header style on table of content page
\fancypagestyle{toc}{
  \fancyhf{}
  \renewcommand{\headrulewidth}{0.4pt}
  \renewcommand{\footrulewidth}{0.4pt}
  \fancyhead[C]{}
  \fancyhead[L]{\textbf{\title}}
  \fancyfoot[L]{\author}
  \fancyfoot[C]{}
  \fancyfoot[R]{\thepage}
}

%%% header style for normal pages
\fancypagestyle{plain}{
  \fancyhf{}
  \renewcommand{\headrulewidth}{0.4pt}
  \renewcommand{\footrulewidth}{0.4pt}
  \fancyhead[C]{}
  \fancyhead[L]{\textbf{\title \hspace{1mm}-- \thechapter. \leftmark}}
  \fancyfoot[L]{\author}
  \fancyfoot[C]{}
  \fancyfoot[R]{\thepage}
}

%%% simple title page
\newcommand\titlep{
  \pagestyle{empty}
  \begin{center}
    \vspace{30mm}
    {\huge\title}
    \vspace{8mm}

    {\bf\Large\subtitle}
    \vfill
    Vypracoval: \hfill \author

    Instituce: \hfill \institution

    Vedl: \hfill \supervisor
    \vspace{10mm}
  \end{center}
  \newpage

}

%%% page with table of content
\newcommand\tocp{
  \pagenumbering{roman}
  \addtocontents{toc}{\protect\thispagestyle{toc}}\pagestyle{toc}
  \tableofcontents
  \cleardoublepage
  \pagestyle{plain}
  \pagenumbering{arabic}
}
